pipeline {
  agent none
  stages {
    stage('Back-end') {
      agent {
        docker { image 'maven:3.8.1-adoptopenjdk-11' }
      }
      steps {
        sh 'mvn archetype:generate -DgroupId=com.example -DartifactId=java-hello-world -DarchetypeArtifactId=maven-archetype-quickstart -DinteractiveMode=false'
        dir('java-hello-world') {
          sh 'mvn package'
        }
      }
    }
    stage('Front-end') {
      agent {
        docker { image 'node:16-alpine' }
      }
      steps {
        echo 'console.log("Hello, Node.js!");' > app.js
        sh 'node app.js'
      }
    }
  }
}
